CREATE TABLE IF NOT EXISTS Question ( 
  id_question INT NOT NULL AUTO_INCREMENT PRIMARY KEY, 
  id_bonne_reponse INT NOT NULL, 
  nom_question VARCHAR(255) NOT NULL, 
  categorie VARCHAR(255) NOT NULL
);
CREATE TABLE IF NOT EXISTS Reponse ( 
  id_reponse INT NOT NULL AUTO_INCREMENT PRIMARY KEY, 
  id_question INT NOT NULL, 
  nom_reponse VARCHAR(255) NOT NULL ,
  FOREIGN KEY (id_question) REFERENCES Question (id_question)
);
 
CREATE TABLE IF NOT EXISTS Joueur ( 
  id_joueur INT NOT NULL AUTO_INCREMENT PRIMARY KEY, 
  nom_joueur VARCHAR(30), 
  mot_passe VARCHAR(50) NOT NULL, 
  score_science	INT, 
  score_histoire INT, 
  score_sport INT, 
  score_art INT, 
  score_programmation INT, 
  score_geo INT 
);  

(Une fois que tout est créé)
ALTER TABLE question ADD FOREIGN KEY question(id_bonne_reponse) REFERENCES reponse(id_reponse);
